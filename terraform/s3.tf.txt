# STAGE 1: S3 Bucket Misconfiguration (CTF Entrypoint)
# =============================================================

resource "aws_s3_bucket" "vulnerable_bucket" {
  bucket        = "${var.lab_prefix}-public-assets"
  force_destroy = true

  tags = {
    Name  = "CloudSec-Vulnerable-Bucket"
    Stage = "1-S3-Misconfiguration"
  }
}

resource "aws_s3_bucket_public_access_block" "vulnerable_pab" {
  bucket                  = aws_s3_bucket.vulnerable_bucket.id
  block_public_acls       = false
  block_public_policy     = false
  ignore_public_acls      = false
  restrict_public_buckets = false
}

resource "aws_s3_bucket_policy" "public_read" {
  bucket     = aws_s3_bucket.vulnerable_bucket.id
  depends_on = [aws_s3_bucket_public_access_block.vulnerable_pab]

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Sid       = "PublicRead"
      Effect    = "Allow"
      Principal = "*"
      Action    = ["s3:GetObject", "s3:ListBucket"]
      Resource  = [
        "${aws_s3_bucket.vulnerable_bucket.arn}",
        "${aws_s3_bucket.vulnerable_bucket.arn}/*"
      ]
    }]
  })
}
